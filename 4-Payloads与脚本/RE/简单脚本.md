### [[异或]]
```python
# 定义一个包含多个十六进制数值的列表 s
s = [0x21, 0x6, 0x6, 0x16, 0xb, 0x19, 0x2e, 0x65, 0x35, 0x6a, 0x6f, 0x38, 0x36, 0x84, 0x70, 0x3b, 0x39, 0x65, 0x38, 0x35, 0x84, 0x6f, 0x36, 0x3c, 0x6a, 0x38, 0x68, 0x84, 0x66, 0x70, 0x3b, 0x38, 0x6a, 0x36, 0x34]

# 遍历列表 s 中的每个元素
for i in s:
    # 对每个十六进制数字 i 执行以下操作：
    # 1. 将 i 减去 5，得到一个新数值
    # 2. 对这个新数值与 0x52（即十进制的 82）进行异或操作（^ 是异或运算符）
    # 3. 使用 chr() 函数将结果转化为对应的字符，并打印出来
    #    - chr() 会把得到的数字转化为字符（ASCII码）
    # 4. end='' 表示打印时不换行，而是将所有字符按顺序连接在一起
    print(chr((i - 5 ^ 0x52)), end='')

```

```python
# 定义一个十六进制数值的列表 s
s = [0xc, 0xc, 0xc, 0xc, 0xc, 0xc, 0x41, 0x25, 0x32, 0x3c, 0x2c, 0x25, 0x3b, 0x18, 0x2c, 0x36, 0x45, 0x42, 0x2e, 0x42, 0x18, 0x27, 0x27, 0x20, 0x3f]

# 定义密钥字符串 'NSSCTF'
key = 'NSSCTF'

# 遍历列表 s 中的每个元素，使用 i 来表示当前的索引
for i in range(len(s)):
    # 1. 获取列表 s 中第 i 个元素 s[i]
    # 2. 将 s[i] 减去 12。这个操作是对每个数字的偏移调整。
    # 3. 对调整后的结果与 key 中的字符进行异或操作：
    #    - 使用 key[i % 6] 来循环访问密钥中的字符。通过 i % 6 确保每次循环都能正确访问密钥字符。
    #    - ord(key[i % 6]) 获取密钥字符的 ASCII 值。例如，'N' 的 ASCII 值是 78。
    # 4. 将异或后的结果传递给 chr() 函数，将其转换为对应的字符。
    # 5. 使用 end='' 来确保所有字符打印在同一行，不换行。
    print(chr((s[i] - 12) ^ ord(key[i % 6])), end='')
```

### [[AES]]
```python
# 导入 AES 加密模块
from Crypto.Cipher import AES

# 定义一个解密函数 aes_decrypt，接收密钥和密文作为参数
def aes_decrypt(key_str, ciphertext_hex):
    # 将16进制字符串转换为字节序列
    # 使用 Python 内建的 bytes.fromhex() 方法将十六进制字符串转换为字节序列
    ciphertext_bytes = bytes.fromhex(ciphertext_hex)
    
    # 将密钥字符串转换为字节序列
    # 使用 UTF-8 编码将密钥字符串 'key_str' 转换为字节序列
    key = key_str.encode('utf-8')
    
    # 创建 AES 解密器对象
    # 使用 'AES.new()' 方法创建一个 AES 对象。参数是密钥和加密模式。
    # 'AES.MODE_ECB' 指定了 AES 算法的 ECB 模式（电子密码本模式）
    cipher = AES.new(key, AES.MODE_ECB)
    
    # 解密
    # 使用 'decrypt()' 方法对密文字节序列进行解密
    decrypted_data = cipher.decrypt(ciphertext_bytes)
    
    # 打印解密后的数据
    print(decrypted_data)

# 示例
# 定义密钥字符串，密钥为 '1234567890123456'
key_str = '1234567890123456'

# 定义密文，密文是一个16进制字符串
ciphertext_hex = 'F3498AED82CE44E2357C23F5DCF897A43B6A7BFEE0467C591E301CBC38F99913'

# 调用解密函数，并传入密钥和密文
# aes_decrypt() 函数返回解密后的数据，并将其打印到控制台
plaintext_hex_result = aes_decrypt(key_str, ciphertext_hex)
```
